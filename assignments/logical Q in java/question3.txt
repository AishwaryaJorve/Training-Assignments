
--------------------------------------------Question 3--------------------------------------------------------------------


print table of 5 to 10

 
   5    6 .....10
   10 12.....20
    .     .       .
    .     .       .
    50  60....100





---------------------------------------------------Solution-------------------------------------------------------------



public class question3 
{
      public static void main(String[] args) 
      {
		 for(int r=1;r<=10;r++)
		 {
			 for(int c=5;c<=10;c++)
			 {
				 if(r==1)
                                 {
					 System.out.print(r*c+"  ");
				 }

				 else
				 System.out.print(r*c+" ");
			 }
			 System.out.println();
		 }
	  }
}
